# به‌روزرسانی لیست بسته‌ها
sudo apt update

# نصب ابزارهای حیاتی ساخت (مثل GCC/G++، Make) و CMake
sudo apt install build-essential cmake

# نصب ابزارهای توصیه شده (Git برای دریافت سورس کد، Ninja برای کامپایل سریعتر)
sudo apt install git ninja-build

*******************************

# به پوشه اصلی کاربر خود بروید (اگر از قبل آنجا نیستید)
cd ~

# یک پوشه برای پروژه ایجاد و وارد آن شوید
mkdir llvm-project
cd llvm-project

# سورس کد LLVM و MLIR را دانلود کنید
git clone https://github.com/llvm/llvm-project.git
**********************************

# وارد پوشه اصلی پروژه شوید
cd ~/llvm-project

# یک پوشه مجزا برای فایل‌های خروجی کامپایل ایجاد کنید
mkdir build
cd build

# پیکربندی با CMake (فعال کردن MLIR و تنظیم برای ساخت بهینه)
cmake -G Ninja ../llvm-project/llvm \
  -DCMAKE_BUILD_TYPE=Release \
  -DLLVM_ENABLE_PROJECTS="mlir" \
  -DLLVM_ENABLE_RUNTIMES="all"

# شروع فرآیند ساخت با Ninja (این مرحله زمان‌بر است!)
ninja
